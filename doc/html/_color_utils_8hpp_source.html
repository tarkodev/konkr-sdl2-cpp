51 inline SDL_Color brighter(const SDL_Color& c, double f = 0.7) {
52 // Compute minimum intensity to avoid too-small values
53 double r = c.r, g = c.g, b = c.b;
54 double minIntensity = 1.0 / (1.0 - f);
55 if (r < minIntensity) r = minIntensity;
56 if (g < minIntensity) g = minIntensity;
57 if (b < minIntensity) b = minIntensity;
58
59 SDL_Color result;
60 // Scale channels and clamp to [0,255]
61 result.r = static_cast<Uint8>(std::min(255, static_cast<int>(r / f)));
62 result.g = static_cast<Uint8>(std::min(255, static_cast<int>(g / f)));
63 result.b = static_cast<Uint8>(std::min(255, static_cast<int>(b / f)));
64 result.a = c.a; // Preserve alpha
65 return result;
66 }

133 inline SDL_Color fromHtml(const std::string& htmlCode) {
134 if (htmlCode.size() != 7 || htmlCode[0] != '#')

137 SDL_Color color;
