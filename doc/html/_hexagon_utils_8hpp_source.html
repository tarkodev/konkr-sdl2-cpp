40 inline std::pair<int, int> hexRound(double q, double r) {
41 double s = -q - r;
42 int q_round = static_cast<int>(std::round(q));
43 int r_round = static_cast<int>(std::round(r));
44 int s_round = static_cast<int>(std::round(s));
45
46 double q_diff = std::abs(q_round - q);
47 double r_diff = std::abs(r_round - r);
48 double s_diff = std::abs(s_round - s);
49
50 // Adjust the component with the largest rounding error
51 if (q_diff > r_diff && q_diff > s_diff)
52 q_round = -r_round - s_round;
53 else if (r_diff > s_diff)
54 r_round = -q_round - s_round;
55
56 return {q_round, r_round};
57 }
